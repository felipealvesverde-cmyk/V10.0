    }
  }

if (window.SettingsModal && !SettingsModal.__rdRenderFallbackPatched) {
  const __LJ_Settings_render_before_rd_fallback = SettingsModal.render.bind(SettingsModal);
  SettingsModal.render = function() {
    const html = __LJ_Settings_render_before_rd_fallback();
    if (!App.state.showSettingsModal || html.includes('id="rd-settings-panel"')) return html;

    const panel = window.RDSettingsInjection?.renderPanel?.() || '';
    if (!panel) return html;

    if (html.includes('</section>')) {
      return html.replace('</section>', panel + '</section>');
    }

    if (html.includes('</div>')) {
      const idx = html.lastIndexOf('</div>');
      return html.slice(0, idx) + panel + html.slice(idx);
    }

    return html + panel;
  };
  SettingsModal.__rdRenderFallbackPatched = true;
}

  SettingsModal.databasePanelFixed = function() {
    if (typeof this.databasePanel === 'function') return this.databasePanel();
    if (typeof this.localPanel === 'function') return this.localPanel();
    return `<div class="rounded-3xl bg-white border border-slate-100 p-5"><h3 class="text-2xl font-black text-slate-950">Banco de Dados</h3><p class="text-sm text-slate-500 mt-2">Configurações locais preservadas.</p></div>`;

  };

  SettingsModal.render = function() {
    if (!App.state.showSettingsModal) return '';

    const active = this.activeSection();
    let content = '';
    if (active === 'rd') content = this.rdPanelFixed();
    else if (active === 'backup') content = this.backupPanelFixed();
    else content = this.databasePanelFixed();

    return `<div class="fixed inset-0 z-50 bg-slate-950/70 backdrop-blur-sm p-4 overflow-auto">
      <section id="settingsModal" class="max-w-6xl mx-auto rounded-[2rem] bg-slate-50 shadow-2xl overflow-hidden border border-white/20">
        <header class="bg-slate-950 text-white p-6 flex items-start justify-between gap-4">
          <div>
            <div class="inline-flex items-center gap-2 px-3 py-1 rounded-full bg-white/10 text-xs font-black mb-4">
              <i data-lucide="settings" class="w-4 h-4"></i>
              Configurações do sistema
            </div>
            <h2 class="text-3xl font-black">${active === 'rd' ? 'Integrações' : 'Banco de Dados'}</h2>
            <p class="text-slate-300 mt-2">${active === 'rd' ? 'Conecte provedores externos como RD Station.' : 'Escolha Local, Supabase ou Amazon e deixe o LeadScore pronto para sincronizar.'}</p>
          </div>
          <button onclick="Actions.closeSettingsModal()" class="px-5 py-3 rounded-2xl bg-white/10 hover:bg-white/15 font-black flex items-center gap-2">
            <i data-lucide="x" class="w-4 h-4"></i>
            Fechar
          </button>
        </header>

        <main class="grid lg:grid-cols-[260px_1fr] min-h-[620px]">
          <aside class="bg-white border-r border-slate-200 p-5 space-y-3">

            ${this.sectionButton('rd','RD Station','plug-zap')}
            ${this.sectionButton('backup','Backup em breve','archive')}

          <section class="p-5 lg:p-6 overflow-auto">
            ${content}
          </section>
        </main>
      </section>
    </div>`;
  };

  SettingsModal.__rdLayoutFixV1303 = true;
})();
